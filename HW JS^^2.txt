3**.  Написать функцию, которая принимает на вход слово. Задача функции посчитать и вывести в консоль, сколько в слове гласных, и сколько согласных букв.
e.g. function getWordStructure(word)
В консоли: 
Слово (word) состоит из  (число) гласных и (число) согласных букв

Проверки: 'case', 'Case', 'Check-list'

 function getWordStructure(str){
    str = str.toLowerCase().split(``)
    let Vowel = [`a`,`i`,`u`,`o`,`e`]
    let Odd = [`b`, `c`, `d`, `f`, `g`, `h`, `j`, `k`, `l`, `m`, `n`, `p`, `q`, `r`, `s`, `t`, `v`, `w`, `x`, `y`, `z`]
    let VowelsCount = 0
    let OddsCount = 0
    str.forEach(el => {
        if(Vowel.includes(el)) VowelsCount++
        else if (Odd.includes(el)) OddsCount++
    });
    console.log(`Word ${str.join(``)} consist of  ${VowelsCount} Vowels and ${OddsCount} Odds`)
 }
 getWordStructure(`case`)
 getWordStructure(`Case`)
 getWordStructure(`Check-list`)
